This should only be used for a temporary share on an internal network you trust, and even then it’s sketchy. Does the trick though.

# apt-get install samba

# cat > '/etc/samba/smb.conf' <<BIRDISWORD
[tempshare]
comment = files
read only = yes
locking = yes
path = /somewhere
guest ok = yes
BIRDISWORD
