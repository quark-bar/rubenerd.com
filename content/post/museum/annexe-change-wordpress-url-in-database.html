If you’ve changed your URL in nginx/Apache, and need to modify in the database:

USE wp_database;

BEGIN;

UPDATE wp_options 
SET option_value='http://example.com'
WHERE option_name='siteurl';

UPDATE wp_options 
SET option_value='http://example.com'
WHERE option_name='home';

COMMIT;
